DECLARE

  V_PROSPECTCODE  VARCHAR2(20) := 'HMAH25000087633';
  V_DISBURSALID   NUMBER(10) := 1000076770;
  V_PROSPECTID    VARCHAR2(10);
BEGIN

  SELECT PROSPECTID
    INTO V_PROSPECTID
    FROM QC_LOS.QT_LOANDETAIL
   WHERE PROSPECTCODE = V_PROSPECTCODE;
 
  DELETE FROM QC_LOS.QT_TRANCHE_DTL  WHERE PROSPECTID = V_PROSPECTID
  AND DISBURSALID IS NULL  ;

  DELETE FROM QC_LOS.QT_DISBURSAL D
   WHERE D.PROSPECTID = V_PROSPECTID
     AND D.DISBURSALID = V_DISBURSALID;

  DELETE FROM QC_LOS.QT_DISBURSAL_DTL DD
   WHERE DD.DISBURSALID = V_DISBURSALID;

 DELETE FROM QC_LOS.QT_STRUCTURED_TRANCHE WHERE PROSPECTID = V_PROSPECTID
 AND DISBURSAL_STAGE_ID = 1000000210 ;

END;
